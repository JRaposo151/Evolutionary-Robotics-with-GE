        | <body> <sphere_AUX_Body>
        | <body> <sphere_AUX_Body> <Add_L>

<sphere_AUX_Body>::=  sphere_auxiliar_Link_BACK.urdf <Add_B_BACK>
                    | sphere_auxiliar_Link_BOTTOM.urdf <Add_B_BOTTOM>
                    | sphere_auxiliar_Link_FRONT.urdf <Add_B_FRONT>
                    | sphere_auxiliar_Link_LEFT.urdf <Add_B_LEFT>
                    | sphere_auxiliar_Link_RIGHT.urdf <Add_B_RIGHT>
                    | sphere_auxiliar_Link_TOP.urdf <Add_B_TOP>

<sphere_AUX_Links>::= sphere_auxiliar_LIMBS.urdf

<Add_B_BACK>::= <JointBody> <sphere_AUX_Body> <expr>
<Add_B_BOTTOM>::= <JointBody> <sphere_AUX_Body> <expr>
<Add_B_FRONT>::= <JointBody> <sphere_AUX_Body> <expr>

<Add_B_RIGHT>::= <JointBody> <sphere_AUX_Body> <expr>
<Add_B_TOP>::= <JointBody> <sphere_AUX_Body> <expr>

<JointBody>::=    B_joint_revolute.urdf
                | B_joint_continuous.urdf
                | B_joint_fixed.urdf

<JointLimb>::=    L_joint_revolute.urdf
                | L_joint_continuous.urdf
                | L_joint_fixed.urdf
                | L_joint_revolute_horizontal.urdf

<Limbs>::=    <Big_Link>
            | <Small_Link>
            | <Wheel_Link>
            | <Big_Link> <sphere_AUX_Links> <Add_L>
            | <Small_Link> <sphere_AUX_Links> <Add_L>
            | <Wheel_Link> <sphere_AUX_Links> <Add_L>
